\chapter{Hạn chế và Hướng phát triển}

\section{Hạn chế}
\begin{itemize}
    \item HOG và LBP tuy được khai báo, nhưng độ phân biệt trên tệp ảnh bìa đĩa CD Cover chưa đem lại kết quả MAP ấn tượng như mong đợi. Đặc trưng cổ điển vẫn nhạy cảm nhiều vào ánh sáng.
    \item Việc nối 2 vector Color và HOG chưa có khâu điều chỉnh hệ số $\alpha, \beta$ bằng machine learning nên xảy ra tình trạng "giảm" độ chính xác.
    \item Cơ sở dữ liệu hiện tại còn tương đối nhỏ (99 ảnh Train, 10 ảnh Test). Việc xây dựng KD-Tree trên CSDL nhỏ chưa phô diễn tối đa uy lực so với duyệt Brute-force tuyến tính.
\end{itemize}

\section{Hướng phát triển}
\begin{itemize}
    \item Sử dụng PCA (Principal Component Analysis) để giảm chiều dữ liệu của SIFT/ORB Histogram, giúp tiết kiệm dung lượng đĩa và giảm chi phí chạy K-NN.
    \item Thu thập cơ sở dữ liệu lên đến mức 10,000+ ảnh để thấy sức mạnh tuyệt đối của cấu trúc FLANN.
    \item Áp dụng công nghệ học sâu (Deep Learning) từ mạng CNN (ResNet, VGG16) thay cho đặc trưng cục bộ truyền thống nếu cấu hình cho phép.
\end{itemize}
